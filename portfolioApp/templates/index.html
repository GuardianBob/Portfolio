import User

class AboutMe:
    def __init__(self, first_name, last_name, github, linkedin):
        self.first_name = first_name
        self.last_name = last_name
        self.gitHub_link = github
        self.linkedIn_link = linkedin

    def display_info(self):
        print(f"Hi, I'm {self.first_name} {self.last_name}!", "\n" \
            f"Github: {self.gitHub_link}", "\n" f"LinkedIn: {self.linkedIn_link}", \
                "\n" "Hire me!")

jesse = AboutMe("Jesse", "Meyer", "https://github.com/GuardianBob", "https://www.linkedin.com/in/meyerjg/")
jesse.display_info()
